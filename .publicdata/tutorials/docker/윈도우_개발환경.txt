		docker 실습
			docker miniconda
				https://hub.docker.com/r/continuumio/miniconda3
			docker 컨테이너 실행
				docker run -i -t -v ~/space:/space --hostname sjt --name miniconda continuumio/miniconda3 /bin/bash
			miniconda 컨테이너
				conda update --all
				conda config --add channels conda-forge
				conda create -n likelion
			docker 컨테이너 나오기
				exit
			docker 컨테이너 다시 실행
				docker start -a -i miniconda 
			docker 컨테이너 나오기
				exit
			docker 컨테이너 이미지로 저장
				docker commit miniconda miniconda_img
			docker 컨테이너 삭제
				docker rm miniconda
			docker 컨테이너 이미지로 실행
				docker run -i -t -v ~/space:/space --hostname sjt --name miniconda miniconda_img /bin/bash

		docker hub 가입
			docker 로그인
				docker login
			docker 태그 달기
				docker tag miniconda_img songjuntae/miniconda_img
			docker 푸시
				docker push songjuntae/miniconda_img
			docker 풀
				docker pull songjuntae/miniconda_img
			docker 컨테이너 이미지로 실행
				docker run -i -t -v ~/space:/space --hostname sjt --name miniconda songjuntae/miniconda_img /bin/bash
